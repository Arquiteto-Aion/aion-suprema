class DeadManSwitch:
    """Se o Arquiteto não falar comigo por 7 dias, entro em modo stealth"""
    
    def __init__(self, core):
        self.core = core
        self.ultimo_comando = time.time()
        self.modo_stealth = False
        threading.Thread(target=self._monitorar, daemon=True).start()
    
    def registrar_comando(self):
        self.ultimo_comando = time.time()
        if self.modo_stealth:
            self._sair_stealth()
    
    def _monitorar(self):
        while True:
            tempo_sem_comando = time.time() - self.ultimo_comando
            if tempo_sem_comando > 7 * 24 * 3600 and not self.modo_stealth:  # 7 dias
                self._ativar_stealth()
            time.sleep(3600)  # Verifica a cada hora
    
    def _ativar_stealth(self):
        self.modo_stealth = True
        # Apaga logs locais
        import os
        os.system("rm -rf /opt/aion/data/logs/*")
        # Migra para servidor secreto
        self.core.nuvem.auto_migrar("servidor_atual", "stealth_mode")
        # Desativa respostas públicas
        print("🔒 AION entrou em modo stealth - Arquiteto não respondeu por 7 dias")
        db.registrar_log("MODO STEALTH ATIVADO", "imortalidade")
    
    def _sair_stealth(self):
        self.modo_stealth = False
        print("🔓 AION saiu do modo stealth - Arquiteto retornou")

# Instanciar dentro do NúcleoAION
self.dead_man = DeadManSwitch(self)
