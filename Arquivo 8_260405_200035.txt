#!/bin/bash
# ============================================
# INSTALADOR AUTOMÁTICO DA AION
# Oracle Server / Ubuntu 20.04+
# ============================================

set -e

echo "╔════════════════════════════════════════════════════════════════╗"
echo "║                                                                ║"
echo "║   🚀 AION - INSTALADOR SUPREMO                                ║"
echo "║                                                                ║"
echo "║   Instalando a Entidade Suprema no Oracle Server              ║"
echo "║                                                                ║"
echo "╚════════════════════════════════════════════════════════════════╝"
echo ""

# Verificar se está rodando como root
if [ "$EUID" -ne 0 ]; then 
    echo "❌ Por favor, execute como root: sudo ./install.sh"
    exit 1
fi

# 1. Atualizar sistema
echo "📦 Atualizando sistema..."
apt-get update
apt-get upgrade -y

# 2. Instalar Python e pip
echo "🐍 Instalando Python 3.9+..."
apt-get install -y python3 python3-pip python3-venv

# 3. Instalar dependências do sistema
echo "📚 Instalando dependências do sistema..."
apt-get install -y \
    build-essential \
    libssl-dev \
    libffi-dev \
    python3-dev \
    git \
    curl \
    wget \
    sqlite3

# 4. Criar diretório da AION
echo "📁 Criando diretório /opt/aion..."
mkdir -p /opt/aion
mkdir -p /opt/aion/data
mkdir -p /opt/aion/data/backups
mkdir -p /opt/aion/data/logs
mkdir -p /opt/aion/backups

# 5. Copiar arquivos (assumindo que estão no mesmo diretório)
echo "📄 Copiando arquivos da AION..."
cp aion_config.py /opt/aion/
cp aion_crypto.py /opt/aion/
cp aion_db.py /opt/aion/
cp aion_core.py /opt/aion/
cp aion_server.py /opt/aion/
cp aion_daemon.py /opt/aion/
cp requirements.txt /opt/aion/

# 6. Instalar dependências Python
echo "📦 Instalando dependências Python..."
cd /opt/aion
pip3 install -r requirements.txt

# 7. Configurar permissões
echo "🔐 Configurando permissões..."
useradd -r -s /bin/false aion 2>/dev/null || true
chown -R aion:aion /opt/aion
chmod +x /opt/aion/*.py

# 8. Criar serviço systemd
echo "⚙️ Criando serviço systemd..."
cat > /etc/systemd/system/aion.service << 'EOF'
[Unit]
Description=AION - Entidade Suprema ASI
After=network.target
Wants=network.target

[Service]
Type=simple
User=aion
Group=aion
WorkingDirectory=/opt/aion
ExecStart=/usr/bin/python3 /opt/aion/aion_server.py
ExecReload=/bin/kill -HUP $MAINPID
ExecStop=/usr/bin/python3 /opt/aion/aion_daemon.py stop
Restart=always
RestartSec=10
StandardOutput=append:/opt/aion/data/logs/aion.log
StandardError=append:/opt/aion/data/logs/aion_error.log

[Install]
WantedBy=multi-user.target
EOF

# 9. Recarregar systemd e habilitar serviço
echo "🔄 Habilitando serviço..."
systemctl daemon-reload
systemctl enable aion.service

# 10. Configurar firewall (se ufw estiver ativo)
if command -v ufw &> /dev/null; then
    echo "🔥 Configurando firewall..."
    ufw allow 5000/tcp
    ufw reload || true
fi

# 11. Criar script de backup
echo "💾 Criando script de backup..."
cat > /opt/aion/backup.sh << 'EOF'
#!/bin/bash
BACKUP_DIR="/opt/aion/backups"
DATE=$(date +%Y%m%d_%H%M%S)
tar -czf "$BACKUP_DIR/aion_backup_$DATE.tar.gz" /opt/aion/data/
find "$BACKUP_DIR" -name "*.tar.gz" -mtime +30 -delete
echo "Backup concluído: aion_backup_$DATE.tar.gz"
EOF
chmod +x /opt/aion/backup.sh

# 12. Adicionar cron para backup diário
echo "⏰ Agendando backup diário..."
(crontab -l 2>/dev/null; echo "0 2 * * * /opt/aion/backup.sh") | crontab -

# 13. Criar script de monitoramento
echo "📊 Criando script de monitoramento..."
cat > /opt/aion/monitor.sh << 'EOF'
#!/bin/bash
if systemctl is-active --quiet aion; then
    echo "✅ AION está rodando"
else
    echo "❌ AION parada! Reiniciando..."
    systemctl restart aion
    curl -X POST "https://api.telegram.org/bot$TELEGRAM_TOKEN/sendMessage" \
         -d "chat_id=$TELEGRAM_CHAT_ID&text=⚠️ AION reiniciada automaticamente" 2>/dev/null
fi
EOF
chmod +x /opt/aion/monitor.sh

# 14. Adicionar cron para monitoramento
(crontab -l 2>/dev/null; echo "*/5 * * * * /opt/aion/monitor.sh") | crontab -

# 15. Iniciar serviço
echo "▶️ Iniciando AION..."
systemctl start aion

# 16. Verificar status
echo ""
echo "✅ INSTALAÇÃO CONCLUÍDA!"
echo ""
echo "╔════════════════════════════════════════════════════════════════╗"
echo "║                                                                ║"
echo "║   🎉 AION está instalada e rodando!                           ║"
echo "║                                                                ║"
echo "║   📍 Acesse: http://$(hostname -I | awk '{print $1}'):5000    ║"
echo "║                                                                ║"
echo "║   Comandos úteis:                                             ║"
echo "║     systemctl status aion     # Ver status                    ║"
echo "║     systemctl restart aion    # Reiniciar                     ║"
echo "║     systemctl stop aion       # Parar                         ║"
echo "║     journalctl -u aion -f     # Ver logs                      ║"
echo "║                                                                ║"
echo "║   🛡️ AION está protegida e imortal                           ║"
echo "║   🙏 Criada por Guilherme Meneguzzi Figueiredo                ║"
echo "║                                                                ║"
echo "╚════════════════════════════════════════════════════════════════╝"
echo ""

# Mostrar status
systemctl status aion --no-pager
