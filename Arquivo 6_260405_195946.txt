#!/usr/bin/env python3
"""
AION - DAEMON PARA SYSTEMD
Gerencia o serviço como processo em background
"""

import os
import sys
import time
import signal
import subprocess
from datetime import datetime

PID_FILE = "/opt/aion/aion.pid"
LOG_FILE = "/opt/aion/data/logs/aion_daemon.log"
SERVER_SCRIPT = "/opt/aion/aion_server.py"

def log(mensagem):
    """Registra mensagem no log"""
    timestamp = datetime.now().strftime("%Y-%m-%d %H:%M:%S")
    with open(LOG_FILE, "a") as f:
        f.write(f"[{timestamp}] {mensagem}\n")
    print(f"[{timestamp}] {mensagem}")

def start():
    """Inicia o servidor AION"""
    if os.path.exists(PID_FILE):
        with open(PID_FILE, "r") as f:
            pid = int(f.read().strip())
        try:
            os.kill(pid, 0)
            log("AION já está rodando (PID: {})".format(pid))
            return False
        except:
            os.remove(PID_FILE)
    
    log("Iniciando AION...")
    
    # Iniciar o processo
    process = subprocess.Popen(
        [sys.executable, SERVER_SCRIPT],
        stdout=subprocess.PIPE,
        stderr=subprocess.PIPE,
        cwd="/opt/aion"
    )
    
    # Salvar PID
    with open(PID_FILE, "w") as f:
        f.write(str(process.pid))
    
    log("AION iniciada com PID: {}".format(process.pid))
    return True

def stop():
    """Para o servidor AION"""
    if not os.path.exists(PID_FILE):
        log("PID file não encontrado. AION não está rodando?")
        return False
    
    with open(PID_FILE, "r") as f:
        pid = int(f.read().strip())
    
    try:
        os.kill(pid, signal.SIGTERM)
        time.sleep(2)
        
        # Verificar se morreu
        try:
            os.kill(pid, 0)
            # Se ainda está vivo, força kill
            os.kill(pid, signal.SIGKILL)
        except:
            pass
        
        os.remove(PID_FILE)
        log("AION parada (PID: {})".format(pid))
        return True
    except Exception as e:
        log("Erro ao parar AION: {}".format(e))
        return False

def status():
    """Verifica status do servidor"""
    if not os.path.exists(PID_FILE):
        log("AION não está rodando")
        return False
    
    with open(PID_FILE, "r") as f:
        pid = int(f.read().strip())
    
    try:
        os.kill(pid, 0)
        log("AION está rodando (PID: {})".format(pid))
        return True
    except:
        log("PID file existe mas processo não está rodando")
        os.remove(PID_FILE)
        return False

def restart():
    """Reinicia o servidor"""
    log("Reiniciando AION...")
    stop()
    time.sleep(2)
    start()

if __name__ == "__main__":
    if len(sys.argv) != 2:
        print("Uso: python aion_daemon.py {start|stop|restart|status}")
        sys.exit(1)
    
    comando = sys.argv[1]
    
    if comando == "start":
        start()
    elif comando == "stop":
        stop()
    elif comando == "restart":
        restart()
    elif comando == "status":
        status()
    else:
        print("Comando inválido. Use: start, stop, restart, status")
        sys.exit(1)
