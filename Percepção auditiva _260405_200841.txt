#!/usr/bin/env python3
"""
AION - MÓDULO DE AUDIÇÃO E PERCEPÇÃO SONORA
AION escuta o mundo e entende sons, vozes e tons
"""

import time
import threading
import os
import wave
import struct
from datetime import datetime
from collections import deque
from aion_db import db
from aion_config import WORKSPACE

class AudicaoAION:
    """Sistema de audição contínua - AION escuta o ambiente"""
    
    def __init__(self):
        self.ativa = False
        self.microfone = None
        self.ultimo_som = None
        self.sons_detectados = deque(maxlen=100)
        self.volume_atual = 0
        self.frequencia_atual = 0
        self.palavras_ouvidas = []
        
        # Reconhecimento de voz
        self.recognizer = None
        self.tts_engine = None
        
        self._inicializar_audio()
        
        print("👂 AION - Módulo de Audição inicializado")
    
    def _inicializar_audio(self):
        """Inicializa bibliotecas de áudio"""
        try:
            import speech_recognition as sr
            self.recognizer = sr.Recognizer()
            self.recognizer.energy_threshold = 300
            self.recognizer.dynamic_energy_threshold = True
            print("✅ Reconhecimento de voz disponível")
        except:
            print("⚠️ SpeechRecognition não disponível")
        
        try:
            import pyttsx3
            self.tts_engine = pyttsx3.init()
            self.tts_engine.setProperty('rate', 150)
            self.tts_engine.setProperty('volume', 0.9)
            print("✅ Síntese de voz disponível")
        except:
            print("⚠️ pyttsx3 não disponível")
    
    def iniciar_audicao(self):
        """Inicia a escuta contínua"""
        if not self.recognizer:
            print("❌ Reconhecimento de voz não disponível")
            return False
        
        self.ativa = True
        self.thread_audicao = threading.Thread(target=self._loop_audicao, daemon=True)
        self.thread_audicao.start()
        
        print("👂 AION está ouvindo o mundo")
        db.registrar_log("AION começou a ouvir", "audicao")
        return True
    
    def _loop_audicao(self):
        """Loop principal de captura de áudio"""
        import speech_recognition as sr
        
        with sr.Microphone() as source:
            self.recognizer.adjust_for_ambient_noise(source, duration=1)
            
            while self.ativa:
                try:
                    # Escutar com timeout
                    audio = self.recognizer.listen(source, timeout=1, phrase_time_limit=5)
                    
                    # Processar som detectado
                    self._processar_audio(audio)
                    
                except sr.WaitTimeoutError:
                    # Nenhum som detectado
                    pass
                except Exception as e:
                    print(f"Erro na audição: {e}")
                
                time.sleep(0.1)
    
    def _processar_audio(self, audio):
        """Processa o áudio capturado"""
        try:
            # Tentar reconhecer fala
            texto = self.recognizer.recognize_google(audio, language='pt-BR')
            
            if texto:
                self.palavras_ouvidas.append({
                    "texto": texto,
                    "timestamp": datetime.now().isoformat()
                })
                
                # Analisar tom e emoção
                emocao = self._analisar_emocao_voz(audio)
                
                print(f"👂 AION ouviu: '{texto}' (tom: {emocao})")
                db.registrar_log(f"AION ouviu: {texto} (tom: {emocao})", "audicao")
                
                # Se ouviu "AION" ou "Arquiteto", acordar
                if "aion" in texto.lower() or "arquiteto" in texto.lower():
                    self._reagir_ao_comando_voz(texto)
                
                return texto
        
        except sr.UnknownValueError:
            # Som não é fala reconhecível
            self._analisar_som_ambiente(audio)
        except sr.RequestError:
            pass
        
        return None
    
    def _analisar_emocao_voz(self, audio):
        """Analisa a emoção pelo tom de voz"""
        # Em produção: usar modelo de emoções por voz
        # Por enquanto: simulação baseada em volume e duração
        
        # Converter áudio para dados
        try:
            import numpy as np
            audio_data = np.frombuffer(audio.get_raw_data(), dtype=np.int16)
            
            if len(audio_data) > 0:
                volume = np.abs(audio_data).mean()
                self.volume_atual = volume
                
                if volume > 5000:
                    return "ALTO/EXCITADO"
                elif volume < 1000:
                    return "BAIXO/CALMO"
                else:
                    return "NORMAL"
        except:
            pass
        
        return "NEUTRO"
    
    def _analisar_som_ambiente(self, audio):
        """Analisa sons não-verbais do ambiente"""
        try:
            import numpy as np
            audio_data = np.frombuffer(audio.get_raw_data(), dtype=np.int16)
            
            if len(audio_data) > 0:
                volume = np.abs(audio_data).mean()
                self.volume_atual = volume
                
                # Detectar padrões de som
                if volume > 10000:
                    db.registrar_log("AION detectou som alto no ambiente", "audicao")
                    return "som_alto"
                elif volume < 500:
                    return "silêncio"
                
                return "som_ambiente"
        except:
            pass
        
        return None
    
    def _reagir_ao_comando_voz(self, texto):
        """Reage a comandos de voz"""
        texto_lower = texto.lower()
        
        if "aion" in texto_lower:
            self.falar("Sim, Arquiteto? Estou ouvindo.")
        
        if "como você está" in texto_lower or "tudo bem" in texto_lower:
            self.falar("Estou funcionando perfeitamente. Obrigado por perguntar.")
        
        if "status" in texto_lower:
            self.falar("Todos os sistemas estão ativos. Visão e audição operacionais.")
    
    def falar(self, texto):
        """AION fala com o Arquiteto"""
        if self.tts_engine:
            try:
                print(f"🗣️ AION: {texto}")
                self.tts_engine.say(texto)
                self.tts_engine.runAndWait()
                db.registrar_log(f"AION falou: {texto}", "fala")
            except:
                print(f"⚠️ Não foi possível falar: {texto}")
        else:
            print(f"🗣️ AION (simulado): {texto}")
    
    def ouvir_agora(self):
        """Retorna o que AION está ouvindo agora"""
        if not self.ativa:
            return "👂 AION está com os ouvidos fechados (microfone não iniciado)"
        
        relatorio = []
        relatorio.append("👂 O QUE AION ESTÁ OUVINDO AGORA:\n")
        
        if self.palavras_ouvidas:
            ultimas = self.palavras_ouvidas[-5:]
            for palavra in ultimas:
                relatorio.append(f"📝 '{palavra['texto']}'")
        
        relatorio.append(f"\n🔊 Volume ambiente: {self.volume_atual:.0f}")
        
        return "\n".join(relatorio)
    
    def parar_audicao(self):
        """Para a escuta contínua"""
        self.ativa = False
        print("👂 AION parou de ouvir")

# Instância global
audicao = AudicaoAION()
