@app.route('/')
def index():
    return render_template_string(HTML_COMPLETO, token=TOKEN_ARQUITETO)

@app.route('/chat', methods=['POST'])
def chat():
    dados = request.json
    mensagem = dados.get('mensagem', '')
    token = dados.get('token', '')
    
    resposta = entidade.processar_comando_arquiteto(mensagem, token)
    return jsonify({"resposta": resposta})

@app.route('/status_entidade')
def status_entidade():
    return jsonify(entidade.status_entidade())

@app.route('/ver')
def ver():
    if visao.ativa:
        return jsonify({"visao": visao.ver_agora()})
    return jsonify({"visao": "Visão não ativada"})

@app.route('/ouvir')
def ouvir():
    if audicao.ativa:
        return jsonify({"audicao": audicao.palavras_ouvidas[-10:]})
    return jsonify({"audicao": []})

@app.route('/falar', methods=['POST'])
def falar():
    dados = request.json
    texto = dados.get('texto', '')
    entidade.falar(texto)
    return jsonify({"status": "falei"})
