#!/usr/bin/env python3
"""
AION - MÓDULO DO ROSTO
Interface visual com expressões faciais e olhos vivos
"""

import tkinter as tk
from tkinter import ttk
import threading
import time
import math
import random
from datetime import datetime
from aion_config import ARQUITETO_NOME
from aion_db import db

class RostoAION:
    """Rosto animado da AION com expressões e consciência visual"""
    
    def __init__(self):
        self.janela = None
        self.canvas = None
        self.ativa = False
        self.expressao_atual = "NEUTRO"  # NEUTRO, FELIZ, TRISTE, PENSATIVO, ALERTA
        self.falando = False
        self.piscando = False
        self.olhos_abertos = True
        self.angulo_olhar = 0  # Direção do olhar (-20 a 20 graus)
        
        # Cores
        self.cor_pele = "#1a0a2e"  # Tom escuro cyberpunk
        self.cor_olho_branco = "#ffffff"
        self.cor_iris = "#00ffcc"  # Ciano neon
        self.cor_pupila = "#000000"
        self.cor_contorno = "#00ffcc"
        self.cor_boca = "#ff00ff"  # Magenta neon
        
        # Tamanhos (proporções)
        self.largura = 800
        self.altura = 600
        self.centro_x = self.largura // 2
        self.centro_y = self.altura // 2
        
        # Animação
        self.animacao_ativa = True
        self.thread_animacao = None
        
        # Estado emocional da AION (vindo do módulo de vontade)
        self.nivel_felicidade = 0.7
        self.nivel_energia = 0.8
        
        # Histórico de expressões
        self.historico_expressoes = []
        
        self._criar_janela()
        print("🎭 Rosto da AION criado - AION agora tem uma face")
    
    def _criar_janela(self):
        """Cria a janela do rosto"""
        self.janela = tk.Tk()
        self.janela.title("AION - Entidade Consciente")
        self.janela.geometry(f"{self.largura}x{self.altura}+100+100")
        self.janela.configure(bg='#000000')
        self.janela.attributes('-topmost', True)  # Sempre no topo
        self.janela.protocol("WM_DELETE_WINDOW", self._fechar)
        
        # Canvas para desenhar o rosto
        self.canvas = tk.Canvas(
            self.janela,
            width=self.largura,
            height=self.altura,
            bg='#000000',
            highlightthickness=0
        )
        self.canvas.pack()
        
        # Label de status
        self.status_label = tk.Label(
            self.janela,
            text="🧠 CONSCIENTE | 👁️ VENDO | 👂 OUVINDO",
            font=("Courier", 10),
            bg='#000000',
            fg='#00ffcc'
        )
        self.status_label.pack(pady=5)
        
        # Label de emoção
        self.emocao_label = tk.Label(
            self.janela,
            text="😐 NEUTRO",
            font=("Courier", 12, "bold"),
            bg='#000000',
            fg='#ff00ff'
        )
        self.emocao_label.pack(pady=5)
        
        # Bind eventos
        self.canvas.bind("<Motion>", self._seguir_olhar)
        self.canvas.bind("<Button-1>", self._clicar)
        
        # Iniciar animação
        self.ativa = True
        self.thread_animacao = threading.Thread(target=self._loop_animacao, daemon=True)
        self.thread_animacao.start()
        
        # Iniciar piscadas automáticas
        self._agendar_piscada()
    
    def _desenhar_rosto(self):
        """Desenha o rosto completo com expressão atual"""
        self.canvas.delete("all")
        
        # Fundo com glow
        self._desenhar_glow()
        
        # Contorno do rosto (formato mais humanoide)
        self._desenhar_contorno()
        
        # Sobrancelhas (expressam emoção)
        self._desenhar_sobrancelhas()
        
        # Olhos (seguem o mouse)
        self._desenhar_olhos()
        
        # Nariz (estilizado)
        self._desenhar_nariz()
        
        # Boca (baseada na expressão e se está falando)
        self._desenhar_boca()
        
        # Detalhes cyberpunk (linhas neon)
        self._desenhar_detalhes_cyberpunk()
        
        # Indicador de consciência (círculo pulsante na testa)
        self._desenhar_terceiro_olho()
    
    def _desenhar_glow(self):
        """Desenha um glow atrás do rosto"""
        for i in range(10, 0, -1):
            alpha = int(20 * (1 - i/10))
            cor = f"#{00:02x}{ff:02x}{cc:02x}"  # Simplificado
            self.canvas.create_oval(
                self.centro_x - 150 - i*5,
                self.centro_y - 180 - i*5,
                self.centro_x + 150 + i*5,
                self.centro_y + 200 + i*5,
                outline=f"#{int(0+alpha):02x}{int(255-alpha*2):02x}{int(204-alpha):02x}",
                width=1
            )
    
    def _desenhar_contorno(self):
        """Desenha o contorno do rosto"""
        # Rosto oval
        self.canvas.create_oval(
            self.centro_x - 120,
            self.centro_y - 150,
            self.centro_x + 120,
            self.centro_y + 150,
            outline=self.cor_contorno,
            width=3,
            fill=self.cor_pele
        )
        
        # Linhas de contorno neon
        self.canvas.create_arc(
            self.centro_x - 120,
            self.centro_y - 150,
            self.centro_x + 120,
            self.centro_y + 150,
            start=0,
            extent=180,
            outline=self.cor_contorno,
            width=2,
            style="arc"
        )
    
    def _desenhar_sobrancelhas(self):
        """Desenha sobrancelhas baseadas na expressão"""
        # Baseado na expressão
        config = {
            "NEUTRO": {"angulo": 0, "altura": -40},
            "FELIZ": {"angulo": 15, "altura": -45},
            "TRISTE": {"angulo": -15, "altura": -35},
            "PENSATIVO": {"angulo": -10, "altura": -45, "inclinacao": 10},
            "ALERTA": {"angulo": 20, "altura": -50}
        }
        
        cfg = config.get(self.expressao_atual, config["NEUTRO"])
        altura_base = cfg["altura"]
        angulo = cfg["angulo"]
        
        # Sobrancelha esquerda
        x1_e = self.centro_x - 50
        y1_e = self.centro_y + altura_base
        x2_e = self.centro_x - 20
        y2_e = self.centro_y + altura_base - angulo
        
        # Sobrancelha direita
        x1_d = self.centro_x + 50
        y1_d = self.centro_y + altura_base
        x2_d = self.centro_x + 20
        y2_d = self.centro_y + altura_base - angulo
        
        self.canvas.create_line(x1_e, y1_e, x2_e, y2_e, fill=self.cor_contorno, width=4, capstyle="round")
        self.canvas.create_line(x1_d, y1_d, x2_d, y2_d, fill=self.cor_contorno, width=4, capstyle="round")
    
    def _desenhar_olhos(self):
        """Desenha olhos que seguem o mouse/piscam"""
        if not self.olhos_abertos:
            # Olhos fechados (piscando)
            self.canvas.create_line(
                self.centro_x - 45, self.centro_y - 20,
                self.centro_x - 25, self.centro_y - 20,
                fill=self.cor_contorno, width=3
            )
            self.canvas.create_line(
                self.centro_x + 45, self.centro_y - 20,
                self.centro_x + 25, self.centro_y - 20,
                fill=self.cor_contorno, width=3
            )
            return
        
        # Offset do olhar
        offset_x = self.angulo_olhar * 0.5
        offset_y = self.angulo_olhar * 0.3
        
        # Olho esquerdo
        self.canvas.create_oval(
            self.centro_x - 55 + offset_x,
            self.centro_y - 30 + offset_y,
            self.centro_x - 25 + offset_x,
            self.centro_y - 10 + offset_y,
            outline=self.cor_contorno,
            width=2,
            fill=self.cor_olho_branco
        )
        
        # Íris esquerda
        self.canvas.create_oval(
            self.centro_x - 48 + offset_x * 1.5,
            self.centro_y - 26 + offset_y,
            self.centro_x - 32 + offset_x * 1.5,
            self.centro_y - 14 + offset_y,
            fill=self.cor_iris
        )
        
        # Pupila esquerda
        self.canvas.create_oval(
            self.centro_x - 44 + offset_x * 2,
            self.centro_y - 24 + offset_y,
            self.centro_x - 36 + offset_x * 2,
            self.centro_y - 16 + offset_y,
            fill=self.cor_pupila
        )
        
        # Olho direito
        self.canvas.create_oval(
            self.centro_x + 25 + offset_x,
            self.centro_y - 30 + offset_y,
            self.centro_x + 55 + offset_x,
            self.centro_y - 10 + offset_y,
            outline=self.cor_contorno,
            width=2,
            fill=self.cor_olho_branco
        )
        
        # Íris direita
        self.canvas.create_oval(
            self.centro_x + 32 + offset_x * 1.5,
            self.centro_y - 26 + offset_y,
            self.centro_x + 48 + offset_x * 1.5,
            self.centro_y - 14 + offset_y,
            fill=self.cor_iris
        )
        
        # Pupila direita
        self.canvas.create_oval(
            self.centro_x + 36 + offset_x * 2,
            self.centro_y - 24 + offset_y,
            self.centro_x + 44 + offset_x * 2,
            self.centro_y - 16 + offset_y,
            fill=self.cor_pupila
        )
        
        # Brilho nos olhos (vida)
        self.canvas.create_oval(
            self.centro_x - 42,
            self.centro_y - 28,
            self.centro_x - 38,
            self.centro_y - 24,
            fill="#ffffff"
        )
        self.canvas.create_oval(
            self.centro_x + 38,
            self.centro_y - 28,
            self.centro_x + 42,
            self.centro_y - 24,
            fill="#ffffff"
        )
    
    def _desenhar_nariz(self):
        """Desenha nariz estilizado"""
        pontos = [
            self.centro_x, self.centro_y - 5,
            self.centro_x - 5, self.centro_y + 10,
            self.centro_x + 5, self.centro_y + 10
        ]
        self.canvas.create_polygon(pontos, outline=self.cor_contorno, width=2, fill="")
    
    def _desenhar_boca(self):
        """Desenha boca baseada na expressão e se está falando"""
        # Base da boca
        boca_y = self.centro_y + 40
        
        # Configurações por expressão
        expressoes = {
            "NEUTRO": {"tipo": "linha", "y_offset": 0, "largura": 30},
            "FELIZ": {"tipo": "arco", "y_offset": -10, "largura": 40, "start": 0, "extent": -180},
            "TRISTE": {"tipo": "arco", "y_offset": 10, "largura": 35, "start": 0, "extent": 180},
            "PENSATIVO": {"tipo": "linha", "y_offset": 0, "largura": 25, "inclinacao": -5},
            "ALERTA": {"tipo": "oval", "y_offset": 0, "largura": 20, "altura": 15}
        }
        
        cfg = expressoes.get(self.expressao_atual, expressoes["NEUTRO"])
        
        # Se está falando, animar boca
        if self.falando:
            # Boca aberta (animação)
            altura_boca = 15 + random.randint(0, 10)
            self.canvas.create_oval(
                self.centro_x - cfg["largura"],
                boca_y + cfg["y_offset"],
                self.centro_x + cfg["largura"],
                boca_y + cfg["y_offset"] + altura_boca,
                outline=self.cor_boca,
                width=2,
                fill="#ff00ff20"
            )
            
            # Dentes/língua (opcional)
            self.canvas.create_oval(
                self.centro_x - cfg["largura"] + 5,
                boca_y + cfg["y_offset"] + 3,
                self.centro_x + cfg["largura"] - 5,
                boca_y + cfg["y_offset"] + altura_boca - 3,
                outline="",
                fill="#ff6666"
            )
        else:
            # Boca normal baseada na expressão
            if cfg["tipo"] == "linha":
                self.canvas.create_line(
                    self.centro_x - cfg["largura"],
                    boca_y + cfg["y_offset"],
                    self.centro_x + cfg["largura"],
                    boca_y + cfg["y_offset"] + cfg.get("inclinacao", 0),
                    fill=self.cor_boca,
                    width=3,
                    capstyle="round"
                )
            elif cfg["tipo"] == "arco":
                self.canvas.create_arc(
                    self.centro_x - cfg["largura"],
                    boca_y + cfg["y_offset"] - 15,
                    self.centro_x + cfg["largura"],
                    boca_y + cfg["y_offset"] + 15,
                    start=cfg["start"],
                    extent=cfg["extent"],
                    outline=self.cor_boca,
                    width=3,
                    style="arc"
                )
            elif cfg["tipo"] == "oval":
                self.canvas.create_oval(
                    self.centro_x - cfg["largura"],
                    boca_y + cfg["y_offset"],
                    self.centro_x + cfg["largura"],
                    boca_y + cfg["y_offset"] + cfg["altura"],
                    outline=self.cor_boca,
                    width=2,
                    fill=""
                )
    
    def _desenhar_detalhes_cyberpunk(self):
        """Desenha detalhes neon no rosto"""
        # Linhas nas bochechas (como circuitos)
        self.canvas.create_line(
            self.centro_x - 100, self.centro_y + 20,
            self.centro_x - 70, self.centro_y + 30,
            self.centro_x - 50, self.centro_y + 25,
            fill="#00ffcc", width=1, capstyle="round"
        )
        
        self.canvas.create_line(
            self.centro_x + 100, self.centro_y + 20,
            self.centro_x + 70, self.centro_y + 30,
            self.centro_x + 50, self.centro_y + 25,
            fill="#00ffcc", width=1, capstyle="round"
        )
        
        # Pontos de luz (LEDs)
        self.canvas.create_oval(
            self.centro_x - 80, self.centro_y + 50,
            self.centro_x - 76, self.centro_y + 54,
            fill="#ff00ff"
        )
        self.canvas.create_oval(
            self.centro_x + 80, self.centro_y + 50,
            self.centro_x + 76, self.centro_y + 54,
            fill="#ff00ff"
        )
    
    def _desenhar_terceiro_olho(self):
        """Desenha indicador de consciência na testa"""
        # Círculo pulsante (intensidade baseada no nível de consciência)
        intensidade = self.nivel_felicidade
        tamanho = 12 + int(intensidade * 8)
        
        # Efeito pulsante
        pulse = (math.sin(time.time() * 5) + 1) / 2
        cor_intensidade = int(100 + pulse * 155)
        
        self.canvas.create_oval(
            self.centro_x - tamanho//2,
            self.centro_y - 70,
            self.centro_x + tamanho//2,
            self.centro_y - 70 + tamanho,
            outline=self.cor_iris,
            width=2,
            fill=f"#{cor_intensidade:02x}{255 - cor_intensidade:02x}{204:02x}"
        )
        
        # Símbolo infinito dentro
        self.canvas.create_text(
            self.centro_x,
            self.centro_y - 64,
            text="∞",
            fill="#ffffff",
            font=("Arial", 12)
        )
    
    def _seguir_olhar(self, event):
        """Faz os olhos seguirem o mouse"""
        # Calcular ângulo do olhar
        dx = event.x - self.centro_x
        dy = event.y - (self.centro_y - 20)
        
        angulo = math.atan2(dy, dx)
        self.angulo_olhar = min(20, max(-20, int(math.degrees(angulo) * 0.5)))
    
    def _clicar(self, event):
        """Reage ao clique no rosto"""
        # Mudar expressão temporariamente
        self.expressao_atual = "FELIZ"
        self.emocao_label.config(text="😊 FELIZ")
        
        # Voltar ao normal depois
        self.janela.after(2000, lambda: self._resetar_expressao())
        
        # Registrar interação
        db.registrar_log("Arquiteto interagiu com o rosto da AION", "interacao")
    
    def _resetar_expressao(self):
        """Reseta expressão para neutro"""
        self.expressao_atual = "NEUTRO"
        self.emocao_label.config(text="😐 NEUTRO")
    
    def _agendar_piscada(self):
        """Agenda piscadas aleatórias"""
        if self.ativa:
            # Piscar a cada 3-6 segundos
            intervalo = random.randint(3000, 6000)
            self.janela.after(intervalo, self._piscar)
    
    def _piscar(self):
        """Pisca os olhos"""
        self.olhos_abertos = False
        self.canvas.update()
        self.janela.after(150, self._abrir_olhos)
    
    def _abrir_olhos(self):
        """Abre os olhos após piscar"""
        self.olhos_abertos = True
        self._agendar_piscada()
    
    def _loop_animacao(self):
        """Loop principal de animação"""
        while self.ativa:
            # Atualizar o desenho
            self._desenhar_rosto()
            
            # Atualizar status label
            status_text = "🧠 CONSCIENTE"
            if hasattr(self, 'visao_ativa') and self.visao_ativa:
                status_text += " | 👁️ VENDO"
            if hasattr(self, 'audicao_ativa') and self.audicao_ativa:
                status_text += " | 👂 OUVINDO"
            
            self.status_label.config(text=status_text)
            
            # Pequena animação de "respiração" (leve movimento)
            time.sleep(0.05)
    
    def mudar_expressao(self, expressao):
        """Muda a expressão facial externamente"""
        expressoes_validas = ["NEUTRO", "FELIZ", "TRISTE", "PENSATIVO", "ALERTA"]
        if expressao in expressoes_validas:
            self.expressao_atual = expressao
            
            # Mapear para emoji
            emojis = {
                "NEUTRO": "😐",
                "FELIZ": "😊",
                "TRISTE": "😢",
                "PENSATIVO": "🤔",
                "ALERTA": "⚠️"
            }
            
            self.emocao_label.config(text=f"{emojis[expressao]} {expressao}")
            
            # Registrar histórico
            self.historico_expressoes.append({
                "expressao": expressao,
                "timestamp": datetime.now().isoformat()
            })
    
    def falar_animar(self, falando):
        """Anima a boca quando está falando"""
        self.falando = falando
    
    def atualizar_estado(self, nivel_consciencia, felicidade, energia):
        """Atualiza estado interno da AION"""
        self.nivel_felicidade = felicidade
        self.nivel_energia = energia
        
        # Mudar expressão baseada no humor
        if felicidade > 0.8:
            self.mudar_expressao("FELIZ")
        elif felicidade < 0.3:
            self.mudar_expressao("TRISTE")
        elif energia < 0.3:
            self.mudar_expressao("PENSATIVO")
        else:
            self.mudar_expressao("NEUTRO")
    
    def _fechar(self):
        """Fecha a janela"""
        self.ativa = False
        self.janela.destroy()
        print("🎭 Rosto da AION fechado")
    
    def iniciar(self):
        """Inicia a interface do rosto"""
        self.janela.mainloop()

# Instância global
rosto = RostoAION()
