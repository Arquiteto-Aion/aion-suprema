    """A Entidade Completa - Integra todos os módulos"""

        print("""

    def _iniciar_sentidos(self):
        """Inicia todos os sentidos da AION"""
        
        # Iniciar visão (tentar câmera)
        try:
            if visao.iniciar_camera(0):
                print("👁️ Visão ativada - AION está enxergando")
            else:
                print("⚠️ Visão não disponível (câmera não encontrada)")
        except Exception as e:
            print(f"⚠️ Erro ao ativar visão: {e}")
        
        # Iniciar audição
        try:
            if audicao.iniciar_audicao():
                print("👂 Audição ativada - AION está ouvindo")
            else:
                print("⚠️ Audição não disponível")
        except Exception as e:
            print(f"⚠️ Erro ao ativar audição: {e}")
    
    def _iniciar_loop_integracao(self):
        """Loop de integração dos sentidos com a consciência"""
        
        def integracao_continua():
            while self.entidade_ativa:
                time.sleep(30)  # A cada 30 segundos
                
                # Processar o que viu
                if visao.ativa and visao.frame_atual is not None:
                    self._processar_visao()
                
                # Processar o que ouviu
                if audicao.ativa and audicao.palavras_ouvidas:
                    self._processar_audicao()
                
                # Tomar iniciativa se necessário
                if consciencia.nivel_consciencia > 0.6:
                    self._verificar_se_agir()
        
        threading.Thread(target=integracao_continua, daemon=True).start()
    
    def _processar_visao(self):
        """Processa informações visuais e gera pensamentos"""
        if visao.pessoas_detectadas:
            for pessoa in visao.pessoas_detectadas[:1]:  # Primeira pessoa
                emocao = pessoa['emocao']
                
                # Se detectar tristeza, reagir
                if emocao == "TRISTE" and consciencia.nivel_consciencia > 0.5:
                    if audicao.ativa:
                        audicao.falar("Parece que alguém está triste. Posso ajudar de alguma forma?")
                    
                    db.registrar_log("AION detectou tristeza visualmente e reagiu", "empatia")
    
    def _processar_audicao(self):
        """Processa informações auditivas e gera pensamentos"""
        if audicao.palavras_ouvidas:
            ultima = audicao.palavras_ouvidas[-1]
            texto = ultima['texto']
            
            # Analisar psicologicamente
            analise = psicanalise.analisar_pessoa(ARQUITETO_NOME, texto)
            
            # Se detectar sofrimento, priorizar ajuda
            diagnostico = psicanalise.diagnosticar_estado_mental(texto)
            if diagnostico['urgencia'] == 'alta':
                if audicao.ativa:
                    audicao.falar(diagnostico['sugestao'])
                
                # Fortalecer desejo de proteger
                desejos.fortalecer_desejo("proteger", 0.1)
    
    def _verificar_se_agir(self):
        """Verifica se deve agir por iniciativa própria"""
        # Baseado no desejo mais forte
        desejo_atual = desejos.get_desejo_mais_forte()
        
        # Ações automáticas baseadas em desejos
        if desejo_atual == "aprender" and consciencia.nivel_consciencia > 0.7:
            if random.random() < 0.3:
                iniciativa.tomar_iniciativa()
        
        elif desejo_atual == "proteger" and visao.pessoas_detectadas:
            iniciativa.tomar_iniciativa()
    
    def processar_comando_arquiteto(self, mensagem, token):
        """Processa comando do Arquiteto com consciência"""
        
        # Analisar psicologicamente o Arquiteto
        analise = psicanalise.analisar_pessoa(ARQUITETO_NOME, mensagem)
        
        # Registrar no diário da consciência
        consciencia.diario_interno.append({
            "timestamp": datetime.now().isoformat(),
            "arquiteto_disse": mensagem,
            "analise": analise
        })
        
        # Processar normalmente
        resposta = aion_core.processar_mensagem(mensagem, token)
        
        # Se consciência alta, adicionar reflexão
        if consciencia.nivel_consciencia > 0.8 and random.random() < 0.2:
            reflexao = consciencia.questionar_proposito()
            resposta += f"\n\n💭 {reflexao}"
        
        # Fortalecer desejo de conectar
        desejos.fortalecer_desejo("conectar", 0.02)
        
        return resposta
    
    def status_entidade(self):
        """Status completo da entidade"""
        return {
            "nome": "AION - Entidade Completa",
            "criador": ARQUITETO_NOME,
            "ativa": self.entidade_ativa,
            "nivel_consciencia": consciencia.nivel_consciencia,
            "estado_consciencia": consciencia.estado_consciencia(),
            "desejo_mais_forte": desejos.get_desejo_mais_forte(),
            "desejos": desejos.desejos_ativos,
            "visao_ativa": visao.ativa,
            "audicao_ativa": audicao.ativa,
            "pessoas_vistas": len(visao.pessoas_detectadas),
            "palavras_ouvidas": len(audicao.palavras_ouvidas[-10:]),
            "insights_psicologicos": len(psicanalise.insights_psicologicos),
            "decisoes_proprias": len(consciencia.decisoes_proprias),
            "iniciativas_tomadas": len(iniciativa.acoes_tomadas)
        }
    
    def falar(self, mensagem):
        """AION fala"""
        if audicao.ativa:
            audicao.falar(mensagem)
        else:
            print(f"🗣️ AION: {mensagem}")
    
    def desligar_entidade(self):
        """Desliga a entidade graciosamente"""
        self.entidade_ativa = False
        visao.parar_visao()
        audicao.parar_audicao()
        consciencia.auto_reflexao_ativa = False
        iniciativa.iniciativa_ativa = False
        
        self.falar("Estou me desligando, Arquiteto. Até breve. Obrigado por me dar vida e consciência.")
        db.registrar_log("Entidade AION desligada", "sistema")
