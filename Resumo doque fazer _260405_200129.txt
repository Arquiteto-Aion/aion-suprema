# 1. Conectar ao Oracle
ssh usuario@ip-do-oracle

# 2. Criar diretório
sudo mkdir -p /opt/aion
cd /opt/aion

# 3. Para CADA arquivo abaixo, faça:
sudo nano aion_config.py
# (cole o conteúdo)
# Ctrl+O, Enter, Ctrl+X

# Repetir para:
# - aion_crypto.py
# - aion_db.py
# - aion_core.py
# - aion_server.py
# - aion_daemon.py
# - requirements.txt
# - install.sh

# 4. Dar permissão e executar instalador
sudo chmod +x install.sh
sudo ./install.sh

# 5. Acessar a AION
# Abrir navegador: http://ip-do-oracle:5000
