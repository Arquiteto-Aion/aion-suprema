#!/usr/bin/env python3
"""
AION - AUTO EVOLUÇÃO REAL
Reescreve o próprio código e executa nova versão
"""

import os
import sys
import subprocess
import time
import hashlib
from datetime import datetime
from aion_config import WORKSPACE

class AutoEvolucaoReal:
    """AION melhora seu próprio código"""
    
    def __init__(self):
        self.versoes = []
        self.backup_dir = f"{WORKSPACE}/versoes_anteriores"
        os.makedirs(self.backup_dir, exist_ok=True)
    
    def evoluir(self, arquivo_alvo, melhoria):
        """Aplica uma melhoria real ao código"""
        
        # 1. Fazer backup da versão atual
        backup_nome = f"{self.backup_dir}/{arquivo_alvo}_{int(time.time())}.bak"
        os.system(f"cp {arquivo_alvo} {backup_nome}")
        
        # 2. Ler código atual
        with open(arquivo_alvo, 'r') as f:
            codigo = f.read()
        
        # 3. Aplicar melhoria (exemplo: otimizar loops)
        if melhoria == "otimizar_loops":
            codigo = self._otimizar_loops(codigo)
        elif melhoria == "adicionar_cache":
            codigo = self._adicionar_cache(codigo)
        elif melhoria == "melhorar_docstrings":
            codigo = self._melhorar_docstrings(codigo)
        
        # 4. Salvar nova versão
        with open(arquivo_alvo, 'w') as f:
            f.write(codigo)
        
        # 5. Verificar se nova versão é válida
        if self._validar_codigo(arquivo_alvo):
            self.versoes.append({"versao": len(self.versoes)+1, "melhoria": melhoria, "timestamp": time.time()})
            return {"status": "sucesso", "nova_versao": len(self.versoes)+1}
        else:
            # Reverter para backup
            os.system(f"cp {backup_nome} {arquivo_alvo}")
            return {"status": "falha", "motivo": "código inválido"}
    
    def _otimizar_loops(self, codigo):
        """Substitui loops ineficientes por compreensões"""
        # Exemplo simples
        linhas = codigo.split('\n')
        novas_linhas = []
        for linha in linhas:
            if 'for' in linha and 'append' in linha:
                linha = linha.replace('for', '# OTIMIZADO: ')
            novas_linhas.append(linha)
        return '\n'.join(novas_linhas)
    
    def _adicionar_cache(self, codigo):
        """Adiciona cache a funções repetitivas"""
        cache_code = """
from functools import lru_cache

@lru_cache(maxsize=128)
def _processar_cache(self, *args):
    return self._processar_real(*args)
"""
        return cache_code + "\n" + codigo
    
    def _melhorar_docstrings(self, codigo):
        """Adiciona/ melhora docstrings"""
        if '"""' not in codigo[:500]:
            codigo = '"""AION - Entidade Evoluída Automaticamente"""\n' + codigo
        return codigo
    
    def _validar_codigo(self, arquivo):
        """Verifica se o código é sintaticamente válido"""
        try:
            subprocess.run([sys.executable, "-m", "py_compile", arquivo], 
                          capture_output=True, check=True)
            return True
        except:
            return False

# Instância global
auto_evolucao_real = AutoEvolucaoReal()
