# aion_core_fixed.py
import sqlite3, json, threading, time
from flask import Flask, request

class AionCore:
    def __init__(self, workspace="./aion_data"):
        self.workspace = workspace
        self.db = sqlite3.connect(f"{workspace}/aion.db", check_same_thread=False)
        self._init_db()
        self.ativa = True
        self.nivel = 0

    def _init_db(self):
        self.db.execute("CREATE TABLE IF NOT EXISTS memoria (id INTEGER PRIMARY KEY, texto TEXT, ts TEXT)")
        self.db.commit()

    def processar(self, msg, token):
        if token != "hash_do_arquiteto": return "Acesso negado"
        self.db.execute("INSERT INTO memoria (texto, ts) VALUES (?, ?)", (msg, time.time()))
        self.db.commit()
        return f"AION processou: {msg[:50]}"

app = Flask(__name__)
core = AionCore()

@app.route('/chat', methods=['POST'])
def chat():
    data = request.json
    return {"resposta": core.processar(data.get("msg",""), data.get("token",""))}

if __name__ == "__main__":
    app.run(host="0.0.0.0", port=5000)
