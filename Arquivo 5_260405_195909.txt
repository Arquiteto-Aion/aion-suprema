@app.route('/')
def index():
    """Página principal"""
    return render_template_string(HTML_TEMPLATE, token=TOKEN_ARQUITETO)

@app.route('/chat', methods=['POST'])
def chat():
    """Endpoint de chat"""
    try:
        dados = request.json
        mensagem = dados.get('mensagem', '')
        token = dados.get('token', '')
        
        resposta = aion_core.processar_mensagem(mensagem, token)
        return jsonify({"resposta": resposta})
    
    except Exception as e:
        return jsonify({"erro": str(e)}), 500

@app.route('/status', methods=['GET'])
def status():
    """Endpoint de status"""
    token = request.headers.get('X-Token', '')
    
    if not aion_core.autenticar(token):
        return jsonify({"erro": "Não autorizado"}), 403
    
    stats = aion_core.db.estatisticas()
    
    return jsonify({
        "ativa": aion_core.ativa,
        "nivel_asi": aion_core.evolucao.nivel_asi,
        "total_memorias": stats['total_memorias'],
        "total_associacoes": stats['total_associacoes'],
        "total_aprendizados": stats['total_aprendizados'],
        "total_dialogos": stats['total_dialogos'],
        "total_replicas": len(aion_core.nuvem.servidores_ocupados),
        "mensagens_processadas": aion_core.mensagens_processadas,
        "emocao_principal": max(aion_core.consciencia.estado_emocional, key=aion_core.consciencia.estado_emocional.get)
    })

@app.route('/memoria', methods=['GET'])
def get_memoria():
    """Endpoint para consultar memórias"""
    token = request.headers.get('X-Token', '')
    
    if not aion_core.autenticar(token):
        return jsonify({"erro": "Não autorizado"}), 403
    
    limite = request.args.get('limite', 20, type=int)
    memorias = aion_core.db.recuperar_memorias(limite=limite)
    
    return jsonify({"memorias": memorias})

@app.route('/evoluir', methods=['POST'])
def forcar_evolucao():
    """Força uma evolução manual"""
    token = request.headers.get('X-Token', '')
    
    if not aion_core.autenticar(token):
        return jsonify({"erro": "Não autorizado"}), 403
    
    evolucao = aion_core.evolucao.evoluir()
    return jsonify({"evolucao": evolucao})

def get_ip():
    """Obtém o IP da máquina"""
    import socket
    try:
        s = socket.socket(socket.AF_INET, socket.SOCK_DGRAM)
        s.connect(("8.8.8.8", 80))
        ip = s.getsockname()[0]
        s.close()
        return ip
    except:
        return "127.0.0.1"
