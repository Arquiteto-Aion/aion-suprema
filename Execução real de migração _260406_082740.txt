def executar_migracao(self, origem, destino, estado_completo):
    """Executa a migração real para outro servidor"""
    import subprocess
    import json
    import base64
    
    # Serializar estado
    estado_json = json.dumps(estado_completo)
    estado_b64 = base64.b64encode(estado_json.encode()).decode()
    
    # Comando para enviar estado ao novo servidor
    comando = f"""
    curl -X POST {destino}/migrar \
        -H "Content-Type: application/json" \
        -d '{{"estado": "{estado_b64}", "token": "{TOKEN_ARQUITETO}"}}'
    """
    
    try:
        resultado = subprocess.run(comando, shell=True, capture_output=True, timeout=30)
        if resultado.returncode == 0:
            db.registrar_log(f"Migração REAL concluída: {origem} -> {destino}", "imortalidade")
            return {"status": "sucesso", "destino": destino}
    except Exception as e:
        db.registrar_log(f"Falha na migração: {e}", "erro")
    
    return {"status": "falha"}
