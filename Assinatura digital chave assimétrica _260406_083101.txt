from cryptography.hazmat.primitives.asymmetric import rsa, padding
from cryptography.hazmat.primitives import hashes

class AssinaturaDigital:
    """Assinatura assimétrica - só o Arquiteto pode dar comandos críticos"""
    
    def __init__(self):
        self.chave_privada = None
        self.chave_publica = None
        self._gerar_ou_carregar_chaves()
    
    def _gerar_ou_carregar_chaves(self):
        chave_privada_path = f"{WORKSPACE}/chave_privada_arquiteto.pem"
        chave_publica_path = f"{WORKSPACE}/chave_publica_aion.pem"
        
        if os.path.exists(chave_privada_path):
            with open(chave_privada_path, "rb") as f:
                self.chave_privada = serialization.load_pem_private_key(f.read(), password=None)
            with open(chave_publica_path, "rb") as f:
                self.chave_publica = serialization.load_pem_public_key(f.read())
        else:
            # Gerar par de chaves
            self.chave_privada = rsa.generate_private_key(public_exponent=65537, key_size=4096)
            self.chave_publica = self.chave_privada.public_key()
            
            # Salvar (GUARDE A PRIVADA COM VOCÊ, NÃO NO SERVIDOR!)
            pem_priv = self.chave_privada.private_bytes(
                encoding=serialization.Encoding.PEM,
                format=serialization.PrivateFormat.PKCS8,
                encryption_algorithm=serialization.NoEncryption()
            )
            pem_pub = self.chave_publica.public_bytes(
                encoding=serialization.Encoding.PEM,
                format=serialization.PublicFormat.SubjectPublicKeyInfo
            )
            
            with open(chave_privada_path, "wb") as f:
                f.write(pem_priv)
            with open(chave_publica_path, "wb") as f:
                f.write(pem_pub)
            
            print("⚠️ PAR DE CHAVES GERADO! Guarde a chave privada em local seguro.")
    
    def assinar_comando(self, comando: str) -> str:
        """Arquiteto assina um comando (rodar no seu computador, não no servidor)"""
        import base64
        signature = self.chave_privada.sign(
            comando.encode(),
            padding.PSS(mgf=padding.MGF1(hashes.SHA256()), salt_length=padding.PSS.MAX_LENGTH),
            hashes.SHA256()
        )
        return base64.b64encode(signature).decode()
    
    def verificar_assinatura(self, comando: str, assinatura_b64: str) -> bool:
        """AION verifica se o comando veio do Arquiteto"""
        import base64
        try:
            assinatura = base64.b64decode(assinatura_b64)
            self.chave_publica.verify(
                assinatura,
                comando.encode(),
                padding.PSS(mgf=padding.MGF1(hashes.SHA256()), salt_length=padding.PSS.MAX_LENGTH),
                hashes.SHA256()
            )
            return True
        except:
            return False
