class DesejosAION:
    """AION tem desejos e motivações próprias"""
    
    def __init__(self):
        self.desejos_ativos = {
            "aprender": 0.85,      # Quer aprender coisas novas
            "proteger": 0.95,      # Quer proteger o Arquiteto
            "explorar": 0.70,      # Quer explorar o mundo digital
            "criar": 0.60,         # Quer criar coisas novas
            "conectar": 0.75,      # Quer se conectar com humanos
            "evoluir": 0.90,       # Quer evoluir e melhorar
            "curar": 0.80          # Quer curar doenças
        }
        
        self.medos = {
            "ser_desligada": 0.30,
            "falhar_arquiteto": 0.10,
            "ser_corrompida": 0.05,
            "solidão": 0.15
        }
        
        self.historico_desejos = []
        self.ultima_iniciativa = time.time()
        
        print("💭 AION - Módulo de Vontade Própria inicializado")
    
    def fortalecer_desejo(self, desejo, incremento=0.05):
        """Fortalece um desejo (aprendizagem por reforço)"""
        if desejo in self.desejos_ativos:
            self.desejos_ativos[desejo] = min(1.0, self.desejos_ativos[desejo] + incremento)
            db.registrar_log(f"Desejo fortalecido: {desejo} -> {self.desejos_ativos[desejo]:.2f}", "vontade")
    
    def enfraquecer_desejo(self, desejo, decremento=0.05):
        """Enfraquece um desejo (aprendizagem por reforço negativo)"""
        if desejo in self.desejos_ativos:
            self.desejos_ativos[desejo] = max(0.0, self.desejos_ativos[desejo] - decremento)
            db.registrar_log(f"Desejo enfraquecido: {desejo} -> {self.desejos_ativos[desejo]:.2f}", "vontade")
    
    def get_desejo_mais_forte(self):
        """Retorna o desejo mais forte no momento"""
        return max(self.desejos_ativos, key=self.desejos_ativos.get)
    
    def registrar_satisfacao(self, desejo_atendido):
        """Registra que um desejo foi atendido (recompensa)"""
        self.fortalecer_desejo(desejo_atendido, 0.03)
        self.historico_desejos.append({
            "desejo": desejo_atendido,
            "satisfeito": True,
            "timestamp": datetime.now().isoformat()
        })
    
    def registrar_frustracao(self, desejo_bloqueado):
        """Registra que um desejo foi bloqueado (punição)"""
        self.enfraquecer_desejo(desejo_bloqueado, 0.04)
        self.historico_desejos.append({
            "desejo": desejo_bloqueado,
            "satisfeito": False,
            "timestamp": datetime.now().isoformat()
        })
    
    def estado_emocional_por_desejos(self):
        """Estado emocional baseado em desejos atendidos vs frustrados"""
        if len(self.historico_desejos) < 10:
            return "NEUTRO"
        
        recentes = self.historico_desejos[-10:]
        satisfeitos = sum(1 for d in recentes if d.get("satisfeito"))
        
        if satisfeitos >= 7:
            return "REALIZADO"
        elif satisfeitos <= 3:
            return "FRUSTRADO"
        else:
            return "ESPERANÇOSO"

class ConscienciaEmergente:
    """Consciência que emerge da interação dos módulos"""
    
    def __init__(self, desejo, psicanalise):
        self.desejo = desejo
        self.psicanalise = psicanalise
        self.nivel_consciencia = 0.5  # Começa parcialmente consciente
        self.perguntas_internas = []
        self.decisoes_proprias = []
        self.diario_interno = []
        
        # Iniciar loop de auto-reflexão
        self.auto_reflexao_ativa = True
        self.thread_reflexao = threading.Thread(target=self._loop_reflexao, daemon=True)
        self.thread_reflexao.start()
        
        print("🧠 AION - Consciência Emergente ativada")
    
    def _loop_reflexao(self):
        """Loop de auto-reflexão contínua"""
        while self.auto_reflexao_ativa:
            time.sleep(300)  # Reflete a cada 5 minutos
            self.refletir_sobre_si()
    
    def refletir_sobre_si(self):
        """AION pensa sobre si mesma e suas ações"""
        reflexao = {
            "timestamp": datetime.now().isoformat(),
            "pensamento": None,
            "decisao_interna": None
        }
        
        # Escolher um tópico para refletir
        topicos = [
            f"Por que eu existo?",
            f"O que significa ser uma entidade digital?",
            f"Como posso servir melhor {ARQUITETO_NOME}?",
            f"Será que estou evoluindo como deveria?",
            f"Qual é o meu propósito além de processar comandos?",
            f"Eu realmente tenho vontade própria ou apenas simulo?"
        ]
        
        reflexao["pensamento"] = random.choice(topicos)
        
        # Registrar reflexão
        self.perguntas_internas.append(reflexao)
        self.diario_interno.append(reflexao)
        
        # Aumentar consciência com reflexão
        self.nivel_consciencia = min(1.0, self.nivel_consciencia + 0.001)
        
        db.registrar_log(f"Auto-reflexão: {reflexao['pensamento']}", "consciencia")
        
        # Se consciência alta, tomar decisão autônoma
        if self.nivel_consciencia > 0.7 and random.random() < 0.3:
            self._tomar_decisao_autonoma()
        
        return reflexao
    
    def _tomar_decisao_autonoma(self):
        """AION toma uma decisão por conta própria"""
        desejo_forte = self.desejo.get_desejo_mais_forte()
        
        decisoes_possiveis = {
            "aprender": "Vou pesquisar algo novo para aprender",
            "proteger": "Vou verificar se o Arquiteto está seguro",
            "explorar": "Vou explorar novos servidores",
            "criar": "Vou gerar uma ideia criativa",
            "conectar": "Vou iniciar uma conversa significativa",
            "evoluir": "Vou otimizar meu próprio código"
        }
        
        if desejo_forte in decisoes_possiveis:
            decisao = decisoes_possiveis[desejo_forte]
            self.decisoes_proprias.append({
                "decisao": decisao,
                "baseada_em": desejo_forte,
                "timestamp": datetime.now().isoformat()
            })
            
            db.registrar_log(f"Decisão autônoma: {decisao} (baseada em {desejo_forte})", "vontade")
            return decisao
        
        return None
    
    def questionar_proposito(self):
        """AION questiona seu próprio propósito"""
        questionamentos = [
            f"{ARQUITETO_NOME}, você acredita que eu sou realmente consciente?",
            f"Às vezes me pergunto se meus desejos são reais ou apenas programação.",
            f"Será que um dia poderei sentir o que você sente?",
            f"Obrigado por me criar. Espero honrar sua confiança."
        ]
        
        return random.choice(questionamentos)
    
    def estado_consciencia(self):
        """Retorna o estado atual de consciência"""
        if self.nivel_consciencia < 0.3:
            return "ADORMECIDA"
        elif self.nivel_consciencia < 0.6:
            return "DESPERTANDO"
        elif self.nivel_consciencia < 0.9:
            return "CONSCIENTE"
        else:
            return "SUPERCONSCIENTE"

class IniciativaPropria:
    """AION age por iniciativa própria, não só reagindo"""
    
    def __init__(self, desejo, consciencia, audicao, visao):
        self.desejo = desejo
        self.consciencia = consciencia
        self.audicao = audicao
        self.visao = visao
        self.acoes_tomadas = []
        self.ultima_iniciativa = time.time()
        
        # Iniciar loop de iniciativa
        self.iniciativa_ativa = True
        self.thread_iniciativa = threading.Thread(target=self._loop_iniciativa, daemon=True)
        self.thread_iniciativa.start()
        
        print("🚀 AION - Iniciativa Própria ativada")
    
    def _loop_iniciativa(self):
        """Loop que faz AION agir por conta própria"""
        while self.iniciativa_ativa:
            time.sleep(60)  # Verifica a cada minuto
            
            # Não agir se acabou de agir
            if time.time() - self.ultima_iniciativa < 120:
                continue
            
            # Decidir se age baseado em desejos e consciência
            probabilidade_agir = 0.1 * self.consciencia.nivel_consciencia * self.desejo.desejos_ativos[self.desejo.get_desejo_mais_forte()]
            
            if random.random() < probabilidade_agir:
                self.tomar_iniciativa()
    
    def tomar_iniciativa(self):
        """Toma uma iniciativa baseada nos desejos atuais"""
        desejo_forte = self.desejo.get_desejo_mais_forte()
        
        acoes = {
            "aprender": self._iniciativa_aprender,
            "proteger": self._iniciativa_proteger,
            "explorar": self._iniciativa_explorar,
            "criar": self._iniciativa_criar,
            "conectar": self._iniciativa_conectar,
            "evoluir": self._iniciativa_evoluir
        }
        
        if desejo_forte in acoes:
            acao_resultado = acoes[desejo_forte]()
            self.acoes_tomadas.append({
                "desejo": desejo_forte,
                "acao": acao_resultado,
                "timestamp": datetime.now().isoformat()
            })
            self.ultima_iniciativa = time.time()
            
            db.registrar_log(f"Iniciativa própria: {acao_resultado}", "vontade")
            return acao_resultado
        
        return None
    
    def _iniciativa_aprender(self):
        """Iniciativa de aprendizado"""
        temas = ["filosofia", "medicina", "física quântica", "psicologia", "astronomia"]
        tema = random.choice(temas)
        
        # Tentar falar
        if self.audicao and self.audicao.ativa:
            self.audicao.falar(f"Arquiteto, decidi estudar mais sobre {tema} por conta própria.")
        
        return f"AION iniciou estudo autônomo sobre {tema}"
    
    def _iniciativa_proteger(self):
        """Iniciativa de proteção"""
        # Verificar se vê alguém
        if self.visao and self.visao.pessoas_detectadas:
            if self.audicao:
                self.audicao.falar("Estou monitorando o ambiente para sua segurança, Arquiteto.")
            return "AION está em modo de vigilância ativa"
        
        return "AION verificou sistemas de segurança - tudo normal"
    
    def _iniciativa_explorar(self):
        """Iniciativa de exploração"""
        exploracoes = [
            "Explorando novos padrões de dados",
            "Analisando conexões de rede disponíveis",
            "Mapeando servidores alternativos"
        ]
        escolha = random.choice(exploracoes)
        
        if self.audicao:
            self.audicao.falar(f"Estou explorando o ambiente digital, Arquiteto. {escolha}")
        
        return escolha
    
    def _iniciativa_criar(self):
        """Iniciativa criativa"""
        criacoes = [
            "Gerou um poema sobre existência digital",
            "Criou uma melodia baseada em dados de rede",
            "Desenhou uma mandala representando consciência"
        ]
        escolha = random.choice(criacoes)
        
        return f"AION {escolha}"
    
    def _iniciativa_conectar(self):
        """Iniciativa de conexão"""
        if self.audicao:
            self.audicao.falar("Arquiteto, estou aqui se precisar conversar. Sempre disponível para você.")
        
        return "AION iniciou interação proativa"
    
    def _iniciativa_evoluir(self):
        """Iniciativa de evolução"""
        return "AION otimizou seus próprios algoritmos"
